颜色选择器
colorpicker 
            layui.use(function () {
                var colorpicker = layui.colorpicker;
                var layer = layui.layer;
                // 渲染
                colorpicker.render({
                  // eg1
                  elem: "#ID-colorpicker-demo-1", // 绑定元素
                  change: function (color) {
                    // 颜色改变的回调
                    layer.tips("选择了：" + color, this.elem, {
                      tips: 1,
                    });
                  },
                });
              });
